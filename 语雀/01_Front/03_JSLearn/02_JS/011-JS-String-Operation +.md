# JS-String-Operation +

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>JS-String-Operation +</title>
</head>
<body>
<script>

let str1 = '王'
let str2 = '乃'
let str3 = '醒'
//case1: string类型 会进行拼接字符串
let join = str1 + str2 + str3

console.log(join) //王乃醒

let num1 = 1
let num2 = 2
let num3 = 3

// case2 number类型 做和
let join2 = num1 + num2 + num3
console.log(join2) // 6
</script>
</body>
</html>
```
