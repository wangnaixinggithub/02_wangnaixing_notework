# C++-Variable-Declaration

```java
#include<string>
using namespace std;

int main(){
    //1.整数 short int long long long
    short number1 = 10;
    int number2 = 10;
    long number3 = 10;
    long long number4 = 10;

    //2.浮点数 float double
    float number5 = 0.111111;
    double number6 = 0.111111;

    //3.字符类型
    char c1 = 'a';
    char c2 = 97; //标识ASCII值对应的字符
    char c3 = '\n'; // 转义字符 表示换行。
    char c4 = '\\'; //表示 \
    
    //4.布尔类型
    bool b1 = true;
    bool b2 = false;

    //5.字符串类型 ，须引入头文件 #include<string>
    string str1 = "王乃醒是大帅哥！";

    return 0;
}
```
