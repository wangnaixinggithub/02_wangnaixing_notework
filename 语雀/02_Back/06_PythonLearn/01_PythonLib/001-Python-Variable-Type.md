# Python-Variable-Type

```python

def main():
    str = '王乃醒'
    print(type(str))

    intNum = 1
    print(type(intNum))

    floatNum = 1.1
    print(type(floatNum))

    boolNum = True
    print(type(boolNum))

    tuple1 = (1, 2, 3)
    print(type(tuple1))

    list1 = [1, 2, 3, 4]
    print(type(list1))

    set1 = {1,2,3,4,5}
    print(type(set1))

    dict = {
        "id": 1,
        "name": "王乃醒",
        "sex": 26,
        "height": 178.8
    }

    print(type(dict))


if __name__ == '__main__':
    main()


```
